uniform samplerRECT samp0 : register(s0);

void main(out float4 ocol0 : COLOR0, in float2 uv0 : TEXCOORD0)
{
  float4 c0 = texRECT(samp0, uv0).rgba;
  float red   = 0.0;
  float green = 0.0;

  if (c0.r < 0.35 || c0.b > 0.35)
  {
	green = c0.g + (c0.b / 2);
  }
  else
  {
	//red = 1 - c0.r + (c0.b / 2);
	red = c0.r + 0.4;
  }
  
  ocol0 = float4(red, green, 0.0, 1.0);
}
